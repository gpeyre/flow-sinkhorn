\begin{array}{@{}r@{\qquad}l@{}}
\textbf{Classical}
&
\begin{aligned}[t]
\KL(p\|q) &:= \sum_i p_i \log\!\frac{p_i}{q_i}
\\
|x|_1 &:= \sum_i |x_i|
\\
\KL(p\|q) &\ge \frac12\,|p-q|_1^2
\end{aligned}
\\[0.8em]
\textbf{Quantum}
&
\begin{aligned}[t]
\KL(P\|Q) &:= \Tr\!\bigl(P(\log P-\log Q)\bigr)
\\
|X|_* &:= |\operatorname{svd}(X)|_1
\\
\KL(P\|Q) &\ge \frac12\,|P-Q|_*^2
\end{aligned}
\end{array}
